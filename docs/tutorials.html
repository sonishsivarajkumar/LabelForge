Getting Started with LabelForge

Installation & Setup

Install LabelForge using pip:

pip install labelforge

For development installations with all dependencies:

git clone https://github.com/sonishsivarajkumar/LabelForge.git
cd LabelForge
pip install -e ".[dev]"

Verify your installation:

python -c "import labelforge; print(labelforge.__version__)"
labelforge --help

Your First Labeling Function

Let's create a simple sentiment analysis labeling function:

from labelforge import lf

@lf(name="positive_keywords")
def positive_sentiment(example):
    """Identifies positive sentiment based on keywords."""
    positive_words = ["excellent", "amazing", "love", "great", "awesome"]
    text = example.text.lower()
    return 1 if any(word in text for word in positive_words) else 0

# Test the function
class Example:
    def __init__(self, text):
        self.text = text

# Test cases
test_examples = [
    Example("This movie is excellent!"),
    Example("I hate this product."),
    Example("The service was amazing.")
]

for ex in test_examples:
    label = positive_sentiment(ex)
    print(f"Text: '{ex.text}' -> Label: {label}")

Tip: Start with simple heuristics. You can always add more sophisticated logic later.

Combining Multiple Labeling Functions

Create multiple labeling functions to capture different aspects:

from labelforge import lf, apply_lfs
import re

@lf(name="positive_keywords")
def lf_positive_keywords(example):
    positive_words = ["excellent", "amazing", "love", "great", "awesome"]
    return 1 if any(word in example.text.lower() for word in positive_words) else 0

@lf(name="negative_keywords") 
def lf_negative_keywords(example):
    negative_words = ["terrible", "awful", "hate", "worst", "horrible"]
    return 1 if any(word in example.text.lower() for word in negative_words) else 0

@lf(name="exclamation_positive")
def lf_exclamation_positive(example):
    # Positive exclamations
    return 1 if re.search(r'[!]{1,3}', example.text) and 'good' in example.text.lower() else 0

@lf(name="question_negative")
def lf_question_negative(example):
    # Questions often indicate problems
    return 1 if '?' in example.text and any(word in example.text.lower() 
                                          for word in ['wrong', 'problem', 'issue']) else 0

# Apply all functions to your data
examples = [
    Example("This product is absolutely amazing!"),
    Example("What's wrong with this service?"),
    Example("I love the new features."),
    Example("This is the worst experience ever.")
]

# Apply labeling functions
lf_output = apply_lfs(examples)
print(f"Applied {lf_output.n_lfs} functions to {lf_output.n_examples} examples")
print(f"Vote matrix shape: {lf_output.votes.shape}")

Training the Label Model

Train a probabilistic model to combine labeling function outputs:

from labelforge import LabelModel

# Initialize and train the label model
label_model = LabelModel(cardinality=2, verbose=True)
label_model.fit(lf_output)

# Generate probabilistic labels
soft_labels = label_model.predict_proba(lf_output)
hard_labels = label_model.predict(lf_output)

print(f"Training completed in {label_model.history_['n_iter']} iterations")
print(f"Final log likelihood: {label_model.history_['log_likelihood'][-1]:.4f}")

# Examine predictions
for i, example in enumerate(examples):
    prob = soft_labels[i, 1]  # Probability of positive class
    pred = hard_labels[i]
    print(f"Text: '{example.text[:50]}...'")
    print(f"  Prediction: {pred} (confidence: {prob:.3f})")
    print()

Pro Tip: The label model automatically learns which functions are more accurate and how they correlate.

Text Classification Deep Dive

Building a News Category Classifier

Let's build a comprehensive news article classifier using multiple labeling strategies:

from labelforge import lf
import re
from collections import Counter

# Sports detection
@lf(name="sports_keywords")
def lf_sports_keywords(example):
    sports_terms = ["football", "basketball", "soccer", "baseball", "tennis", 
                   "score", "game", "match", "player", "team", "championship"]
    text = example.text.lower()
    return "sports" if any(term in text for term in sports_terms) else "other"

# Technology detection  
@lf(name="tech_keywords")
def lf_tech_keywords(example):
    tech_terms = ["software", "computer", "internet", "digital", "AI", 
                  "algorithm", "programming", "data", "cloud", "cybersecurity"]
    text = example.text.lower()
    return "technology" if any(term in text for term in tech_terms) else "other"

# Politics detection
@lf(name="politics_keywords") 
def lf_politics_keywords(example):
    politics_terms = ["government", "election", "president", "congress", 
                     "policy", "vote", "senator", "democracy", "legislation"]
    text = example.text.lower()
    return "politics" if any(term in text for term in politics_terms) else "other"

# Pattern-based detection
@lf(name="financial_patterns")
def lf_financial_patterns(example):
    # Look for financial patterns like stock symbols, currencies
    financial_patterns = [
        r'\$\d+(?:\.\d{2})?(?:\s?(?:million|billion|trillion))?',  # Dollar amounts
        r'\b[A-Z]{1,5}\b.*(?:stock|shares|trading)',  # Stock symbols
        r'(?:NYSE|NASDAQ|Dow Jones)',  # Exchanges
        r'(?:GDP|inflation|interest rate|federal reserve)'
    ]
    text = example.text
    return "finance" if any(re.search(pattern, text, re.IGNORECASE) 
                           for pattern in financial_patterns) else "other"

# Use external models
from transformers import pipeline

# Load a pre-trained classifier
topic_classifier = pipeline("zero-shot-classification", 
                           model="facebook/bart-large-mnli")

@lf(name="transformer_classifier")
def lf_transformer_topic(example):
    candidate_labels = ["sports", "technology", "politics", "finance", "other"]
    result = topic_classifier(example.text, candidate_labels)
    
    # Return the top prediction if confidence is high enough
    if result['scores'][0] > 0.6:
        return result['labels'][0]
    return "other"

Note: Combining rule-based and ML-based labeling functions often produces the best results.

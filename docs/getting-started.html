Installation

labelforge --version

Your First 5 Minutes

Create Your First Labeling Function

Let's start with a simple sentiment analysis example. Copy and paste this code into a Python file: 

from labelforge import lf, Example

# Define a simple labeling function
@lf(name="positive_sentiment")
def detect_positive_sentiment(example):
    """Detect positive sentiment based on keywords."""
    positive_words = ["excellent", "amazing", "love", "great", "awesome", "fantastic"]
    text = example.text.lower()
    
    # Return 1 for positive, 0 for negative/neutral
    return 1 if any(word in text for word in positive_words) else 0

# Test with sample data
test_texts = [
    "This product is absolutely amazing!",
    "I love the new features.",
    "The service was terrible.",
    "It's an okay product, nothing special."
]

# Create examples and test the function
for text in test_texts:
    example = Example(text=text)
    label = detect_positive_sentiment(example)
    print(f"Text: '{text}' -> Label: {label}")

Expected Output:

Text: 'This product is absolutely amazing!' -> Label: 1
Text: 'I love the new features.' -> Label: 1
Text: 'The service was terrible.' -> Label: 0
Text: 'It's an okay product, nothing special.' -> Label: 0

Complete Working Example

Multi-Function Sentiment Analysis

from labelforge import lf, LabelModel, apply_lfs
from labelforge.types import Example

# Define multiple labeling functions
@lf(name="positive_keywords")
def lf_positive_keywords(example):
    positive_words = ["excellent", "amazing", "love", "great", "awesome", "fantastic"]
    return 1 if any(word in example.text.lower() for word in positive_words) else 0

@lf(name="negative_keywords")
def lf_negative_keywords(example):
    negative_words = ["terrible", "awful", "hate", "worst", "horrible", "disgusting"]
    return 1 if any(word in example.text.lower() for word in negative_words) else 0

@lf(name="exclamation_positive")
def lf_exclamation_positive(example):
    # Positive exclamations often indicate enthusiasm
    return 1 if "!" in example.text and len(example.text.split()) > 3 else 0

@lf(name="question_concerns")
def lf_question_concerns(example):
    # Questions about problems might indicate negative sentiment
    concern_words = ["problem", "issue", "wrong", "broken", "error"]
    return 1 if "?" in example.text and any(word in example.text.lower() for word in concern_words) else 0

# Create sample dataset
sample_reviews = [
    "This product is absolutely amazing! I love it!",
    "Terrible quality, worst purchase ever.",
    "What's wrong with this item? It seems broken.",
    "Good value for money, works as expected.",
    "The service was fantastic and exceeded my expectations!",
    "I hate this product, it's completely awful.",
    "Average product, nothing special but does the job.",
    "Why is this so expensive? Not worth the price.",
    "Excellent build quality and great customer service!",
    "This is the worst experience I've ever had."
]

# Convert to Example objects
examples = [Example(text=text) for text in sample_reviews]

# Apply labeling functions
lf_output = apply_lfs(examples)

print(f"Applied {lf_output.n_lfs} labeling functions to {lf_output.n_examples} examples")
print(f"Vote matrix shape: {lf_output.votes.shape}")

# Analyze labeling function performance
coverage = lf_output.coverage()
print("\nLabeling Function Coverage:")
for i, lf_name in enumerate(lf_output.lf_names):
    print(f"  {lf_name}: {coverage[i]:.2f}")

# Train the probabilistic label model
label_model = LabelModel(cardinality=2, verbose=True)
label_model.fit(lf_output)

# Generate final predictions
predictions = label_model.predict_proba(lf_output)
hard_predictions = label_model.predict(lf_output)

print(f"\nModel converged in {label_model.history_['n_iter']} iterations")
print("\nFinal Predictions:")
for i, (example, pred, prob) in enumerate(zip(examples, hard_predictions, predictions)):
    sentiment = "Positive" if pred == 1 else "Negative"
    confidence = max(prob)
    print(f"{i+1:2d}. {sentiment} ({confidence:.3f}): {example.text}")

Key Concepts Demonstrated:

  • • Multiple labeling functions with different strategies
  • • Automatic function registration and application
  • • Coverage analysis to understand function performance
  • • Probabilistic model training with EM algorithm
  • • Confidence scores for final predictions

Conflict Analysis

# Analyze function conflicts
overlap = lf_output.overlap()

print(overlap)
print("Function conflict matrix:")
print(conflicts)

Model Diagnostics

# Get model statistics

print("Learned function accuracies:")
for name, acc in stats['lf_accuracies'].items():
    print(f"  {name}: {acc}")
